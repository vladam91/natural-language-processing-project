/**
 * Dohvata i kopira sadrzaj sistemskog fajla u 
 * niz koji obradjuje i vraca kao povratnu vrednost
 */

function getSystemClipboardInfo() 
{       
    $data = explode("\n", file_get_contents("/proc/meminfo"));
    $meminfo = array();
    foreach ($data as $line) {
        list($key, $val) = explode(":", $line);
        $meminfo[$key] = trim($val);
    }
    return $meminfo;
}

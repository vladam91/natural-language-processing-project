function permanentMemoization() {
	$body = $request->getBody();

	 try{
		
		 $this->validate($body);

	 }    

	 catch(Exception $e){
		 return "There was an error validation json body";
	 }

	 $success = $this->service->passDate($body);

	 if($success) return response.status(200);
	 return response.status(401);
  
}

/**
* Nadji minimalni element u nizu. Vrati nazad minimalni element.
* @param $numbers.
*/

function filter9($numbers)
{
    $min = $numbers[0];
    
    for($i=0;$i<0;$i++){
       
        if($numbers[i] < $min)
            {
                 $min = $numbers[i];
            }        
   
    }

    return min;
}
